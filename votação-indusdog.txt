                        }
                    }, { onlyOnce: true });
                } catch (error) {
                    console.error('Erro ao inicializar estrutura:', error);
                }
            }

            async confirmVote() {
                console.log('confirmVote chamada, selectedOption:', this.selectedOption);
                
                if (!this.selectedOption) {
                    console.log('Nenhuma op√ß√£o selecionada');
                    return;
                }
                
                const confirmBtn = document.getElementById('confirm-btn');
                const syncIndicator = document.getElementById('sync-indicator');
                
                try {
                    console.log('Iniciando processo de vota√ß√£o...');
                    
                    // Desabilita bot√£o durante o processo
                    confirmBtn.disabled = true;
                    confirmBtn.innerHTML = '‚è≥ Computando...';
                    
                    // Atualiza indicador
                    if (syncIndicator) {
                        syncIndicator.innerHTML = '‚è≥ Salvando...<div class="loading-spinner"></div>';
                    }
                    
                    // Converte nome de exibi√ß√£o para chave Firebase v√°lida
                    const firebaseKey = this.getFirebaseKey(this.selectedOption);
                    console.log('Salvando voto para:', this.selectedOption, '(chave Firebase:', firebaseKey + ')');
                    
                    // Transa√ß√£o at√¥mica para garantir consist√™ncia
                    const voteRef = ref(database, `votes/${firebaseKey}`);
                    
                    await runTransaction(voteRef, (currentVotes) => {
                        const newValue = (currentVotes || 0) + 1;
                        console.log('Atualizando votos de', currentVotes, 'para', newValue);
                        return newValue;
                    });

                    await runTransaction(this.totalVotesRef, (currentTotal) => {
                        const newTotal = (currentTotal || 0) + 1;
                        console.log('Atualizando total de', currentTotal, 'para', newTotal);
                        return newTotal;
                    });

                    console.log('Voto salvo com sucesso!');
                    
                    // Sucesso
                    if (syncIndicator) {
                        syncIndicator.innerHTML = '‚úÖ Voto computado!';
                    }
                    
                    setTimeout(() => {
                        this.showResultsScreen();
                    }, 500);
                    
                } catch (error) {
                    console.error('Erro ao votar:', error);
                    this.showError('Erro ao computar voto. Tente novamente.');
                    confirmBtn.disabled = false;
                    confirmBtn.innerHTML = '‚úÖ Confirmar Voto';
                    if (syncIndicator) {
                        syncIndicator.innerHTML = '‚ùå Erro ao salvar';
                    }
                }
            }

            updateResultsDisplay() {
    const resultsContainer = document.getElementById('results-container');
    const totalVotesDisplay = document.getElementById('total-count');
    
    // Atualiza o total de votos se o elemento existir
    if (totalVotesDisplay) {
        totalVotesDisplay.textContent = this.totalVotes.toLocaleString('pt-BR');
    }
    
    // Se n√£o estiver na tela de resultados, n√£o precisa atualizar o container
    if (!resultsContainer) return;
    
    // Ordena os resultados por n√∫mero de votos (decrescente)
    const sortedResults = Object.entries(this.votes)
        .sort(([,a], [,b]) => b - a);
    
    const icons = {
        'Indy': 'ü¶¥',
        'Tur.E': 'üöÄ',
        'Tech': '‚ö°Ô∏è',
    };
    
    resultsContainer.innerHTML = '';
    
    sortedResults.forEach(([name, votes]) => {
        const percentage = this.totalVotes > 0 ? 
            ((votes / this.totalVotes) * 100).toFixed(1) : 0;
        
        const resultItem = document.createElement('div');
        resultItem.className = 'result-item';
        resultItem.innerHTML = `
            <div class="result-name">${icons[name]} ${name}</div>
            <div class="result-stats">
                <span class="result-votes">${votes} votos</span>
                <span class="result-percentage">${percentage}%</span>
            </div>
            <div class="progress-bar">
                <div class="progress-fill" style="width: ${percentage}%"></div>
            </div>
        `;
        
        // Anima√ß√£o de atualiza√ß√£o
        resultItem.classList.add('updated');
        setTimeout(() => {
            resultItem.classList.remove('updated');
        }, 1000);
        
        resultsContainer.appendChild(resultItem);
    });
}
