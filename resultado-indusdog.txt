                this.isConnected = false;

                this.init();

            // Converte chave Firebase para nome de exibi√ß√£o
            getDisplayName(firebaseKey) {
                return this.firebaseToDisplay[firebaseKey] || firebaseKey;
            }

            // Converte nome de exibi√ß√£o para chave Firebase
            getFirebaseKey(displayName) {

            init() {
                const connectedRef = ref(database, '.info/connected');
                onValue(connectedRef, (snapshot) => {
                    this.isConnected = snapshot.val();
                    this.updateConnectionStatus();
                });

                onValue(this.votesRef, (snapshot) => {
                    try {
                        const data = snapshot.val();
                        if (data) {
                            this.previousVotes = { ...this.votes };
                            
                            // Converte dados do Firebase para nomes de exibi√ß√£o
                            const convertedVotes = {};
                            Object.entries(data).forEach(([firebaseKey, votes]) => {
                                const displayName = this.getDisplayName(firebaseKey);
                                convertedVotes[displayName] = votes;
                            });
                            
                            this.votes = { ...this.votes, ...convertedVotes };
                            this.updateDisplay();
                            this.updateConnectionStatus('connected');
                        } else {
                            this.updateDisplay();
                        }
                    } catch (error) {
                        console.error('Erro ao processar votos:', error);
                        this.showError('Erro ao carregar resultados');
                        this.updateConnectionStatus('error');
                    }
                });

                onValue(this.totalVotesRef, (snapshot) => {
                    const total = snapshot.val();
                    this.totalVotes = total || 0;
                    this.updateTotalDisplay();
                    this.updateDisplay();
                });
            }

            updateConnectionStatus(status = null) {
                const statusElement = document.getElementById('firebase-status');
                
                if (status === 'connected' || this.isConnected) {
                    statusElement.innerHTML = '<div class="pulse-dot"></div> Conectado';
                    statusElement.className = 'firebase-status connected';
                    this.clearError();
                } else if (status === 'error') {
                    statusElement.innerHTML = '‚ùå Erro';
                    statusElement.className = 'firebase-status error';
                } else {
                    statusElement.innerHTML = '<span class="loading-spinner"></span> Conectando...';
                    statusElement.className = 'firebase-status connecting';
                }
            }

            updateTotalDisplay() {
                const totalElement = document.getElementById('total-count');
                if (totalElement) {
                    totalElement.textContent = this.totalVotes.toLocaleString('pt-BR');
                }
            }

                const container = document.getElementById('results-container');
                const lastUpdated = document.getElementById('last-updated');

                container.innerHTML = '';

                    const percentage = this.totalVotes > 0 ? 
                        Math.round((votes / this.totalVotes) * 100) : 0;
                    
                    const wasUpdated = this.previousVotes[name] !== votes;
                    const isWinner = index === 0 && this.totalVotes > 0;
                    
                    const item = document.createElement('div');
                    item.className = `result-item ${isWinner ? 'winner' : ''} ${wasUpdated ? 'updated' : ''}`;
                    
                    item.innerHTML = `
                        <div class="position-badge ${index === 0 ? 'first' : ''}">${positions[index]}</div>
                        <div class="center-section">
                            <div class="name-text">${icons[name]} ${name}</div>
                            <div class="progress-bar">
                                <div class="progress-fill"></div>
                            </div>
                        </div>
                        <div class="stats-section">
                            <div class="percentage">${percentage}%</div>
                            <div class="vote-count">${votes} votos</div>
                        </div>
                    `;
                    
                    container.appendChild(item);
                    
                    // Atualiza a barra de progresso ap√≥s inser√ß√£o no DOM
                    const progressFill = item.querySelector('.progress-fill');
                    if (progressFill) {
                        // For√ßa repaint antes de aplicar a largura

                        setTimeout(() => {
                            progressFill.style.width = `${percentage}%`;
                        }, 50);
                    }
                    
                    if (wasUpdated) {
                        setTimeout(() => {
                            item.classList.remove('updated');
                        }, 1000);
                    }
                });
            }

            }
